   {
    es_set temp_flash_attacker event_var(userid)
   }
  }
 }

 {
  if (server_var(hl2_rage) == 1) do
  {
   if (event_var(es_steamid) in server_var(hl2_users)) do
   {
    if (event_var(weapon) != weapon_hegrenade) do

     if (event_var(es_userweapon) != weapon_hegrenade) do
     {

      {
       if (event_var(es_userweapon) != weapon_smokegrenade) do
       {
        if (event_var(es_userweapon) != weapon_c4) do
        {
         if (event_var(es_userweapon) != weapon_knife) do
         {
          if (event_var(es_userweapon) == weapon_awp) then es_rand temp_rage_dmg 457 459
          if (event_var(es_userweapon) != weapon_awp) then es_rand temp_rage_dmg 105 130
          es_set shooter_userid event_var(userid)
          if (event_var(es_userteam) == 2) then es_xset team_filter "#alive#ct"
          if (event_var(es_userteam) == 3) then es_xset team_filter "#alive#t"
          es_xset closest_userid 0
          es_xset closest_distance 0
          es foreach player playerid server_var(team_filter) "es_doblock project_hl2/get_player_distance"
          es damage server_var(closest_userid) server_var(temp_rage_dmg) 1073741826 server_var(shooter_userid) event_var(weapon)
         }
        }
       }
      }
     }
    }
   }
  }
 }

block get_player_distance
{
 es_xset player_distance 0
 es playerget location server_var(playerid) xpl_player_x xpl_player_y xpl_player_z
 es vecmath distance server_var(xpl_bullet_x) server_var(xpl_bullet_y) server_var(xpl_bullet_z) server_var(xpl_player_x) server_var(xpl_player_y) server_var(xpl_player_z) lenght
 ifx false(closest_distance) do
 {
  es_set closest_distance server_var(player_distance)
  es_set closest_userid server_var(playerid)
 }
 if (server_var(player_distance) < server_var(closest_distance)) do
 {
  es_set closest_distance server_var(player_distance)
  es_set closest_userid server_var(playerid)
 }
}
